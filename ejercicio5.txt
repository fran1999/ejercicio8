list = []
while True:
	print('presione 1 para agregar')
	print('presione 2 para ordenar')
	print('presione 3 para ver maximo')
	print('presione 4 para ver le minimo')
	print('presione 5 para ver promedio')
	print('presione 0 para terminar')
	n = input()
	if n == '1' :
		print('ingrese un numero')
		x = int(input())
		list.append(x)
	elif n == '2' :
		list.sort()
		print(list)
	elif n == '3' :
		if len(list) > 0:
			print(max(list))
		else:
			print('la lista esta vacia')
	elif n == '4' :
		if len(list) > 0:
			print(min(list))
		else:
			print('la lista esta vacia')
	elif n == '5' :
		if len(list) > 0:
			print(sum(list)/len(list))
		else:
			print('la lista esta vacia')
	elif n == '0':
		break
